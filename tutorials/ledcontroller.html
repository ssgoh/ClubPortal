main.py (MicroPython)

import network
import socket
import time
from machine import Pin

# Configuration
SSID = 'YOUR_WIFI_NAME'
PASSWORD = 'YOUR_WIFI_PASSWORD'

# Hardware Setup
led_external = Pin(15, Pin.OUT)
led_onboard = Pin('LED', Pin.OUT)

def connect_wifi():
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)
    wlan.connect(SSID, PASSWORD)
    
    print('Connecting to WiFi...')
    while not wlan.isconnected():
        led_onboard.toggle()
        time.sleep(0.5)
    
    led_onboard.on()
    print('Connected! IP:', wlan.ifconfig()[0])
    return wlan.ifconfig()[0]

def web_page(state):
    html = f"""<!DOCTYPE html>
    <html>
    <head>
        <title>Pico W Control</title>
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <style>
            body {{ font-family: sans-serif; text-align: center; background: #0f172a; color: white; padding: 20px; }}
            .btn {{ display: block; width: 100%; max-width: 300px; margin: 20px auto; padding: 20px; 
                   text-decoration: none; border-radius: 10px; font-weight: bold; color: white; font-size: 1.2em; }}
            .on {{ background: #ef4444; }}
            .off {{ background: #334155; }}
        </style>
    </head>
    <body>
        <h1>Pi Python Club</h1>
        <p>LED Status: {state}</p>
        <a href="/on" class="btn on">TURN ON</a>
        <a href="/off" class="btn off">TURN OFF</a>
    </body>
    </html>"""
    return html

def serve():
    ip = connect_wifi()
    addr = socket.getaddrinfo('0.0.0.0', 80)[0][-1]
    s = socket.socket()
    s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    s.bind(addr)
    s.listen(1)
    print('Web Server active on http://' + ip)
    
    while True:
        try:
            cl, addr = s.accept()
            request = cl.recv(1024).decode('utf-8')
            if '/on' in request: led_external.on()
            elif '/off' in request: led_external.off()
            state = "ON" if led_external.value() else "OFF"
            response = web_page(state)
            cl.send('HTTP/1.0 200 OK\r\nContent-type: text/html\r\n\r\n')
            cl.send(response)
            cl.close()
        except: cl.close()

serve()

1. Connect LED long leg to a resistor, then to GP15.

Logic Breakdown

The script initializes a WiFi connection in Station mode, creates a socket listener on port 80, and renders a simple HTML string back to the client whenever a request is received.
