import network
import socket
import time
from machine import Pin

# Configuration
SSID = 'YOUR_WIFI_NAME'
PASSWORD = 'YOUR_WIFI_PASSWORD'

# Hardware Setup
# GP15 is connected to the positive leg (anode) of the LED
led_external = Pin(15, Pin.OUT)
# The onboard LED on Pico W is accessed via 'LED' string
led_onboard = Pin('LED', Pin.OUT)

def connect_wifi():
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)
    wlan.connect(SSID, PASSWORD)
    
    print('Connecting to WiFi...')
    # Blink onboard LED while connecting
    while not wlan.isconnected():
        led_onboard.toggle()
        time.sleep(0.5)
    
    led_onboard.on()
    print('Connected! IP:', wlan.ifconfig()[0])
    return wlan.ifconfig()[0]

def web_page(state):
    # This generates the HTML served to your browser
    html = f"""<!DOCTYPE html>
    <html>
    <head>
        <title>Pico W Control</title>
        <meta name="viewport" content="width=device-width, initial-scale=1">
        <style>
            body {{ font-family: sans-serif; text-align: center; background: #0f172a; color: white; padding: 20px; }}
            .btn {{ display: block; width: 100%; max-width: 300px; margin: 20px auto; padding: 20px; 
                   text-decoration: none; border-radius: 10px; font-weight: bold; color: white; font-size: 1.2em; }}
            .on {{ background: #ef4444; }}
            .off {{ background: #334155; }}
            .status {{ margin-top: 50px; color: #94a3b8; font-size: 0.9em; }}
        </style>
    </head>
    <body>
        <h1>Pi Python Club</h1>
        <div style="background: #1e293b; padding: 20px; border-radius: 15px;">
            <p>LED is currently: <strong style="color: {'#4ade80' if state == 'ON' else '#94a3b8'}">{state}</strong></p>
        </div>
        <a href="/on" class="btn on">TURN LED ON</a>
        <a href="/off" class="btn off">TURN LED OFF</a>
        <p class="status">Raspberry Pi Pico W Web Server</p>
    </body>
    </html>"""
    return html

def serve():
    ip = connect_wifi()
    # Prepare the socket
    addr = socket.getaddrinfo('0.0.0.0', 80)[0][-1]
    
    s = socket.socket()
    s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    s.bind(addr)
    s.listen(1)
    
    print('Web Server active on http://' + ip)
    
    while True:
        try:
            cl, addr = s.accept()
            # Receive the request (max 1024 bytes)
            request = cl.recv(1024).decode('utf-8')
            
            # Simple URL routing
            if '/on' in request:
                led_external.on()
            elif '/off' in request:
                led_external.off()
                
            # Get current state for the UI
            state = "ON" if led_external.value() else "OFF"
            response = web_page(state)
            
            # Send HTTP Headers and Content
            cl.send('HTTP/1.0 200 OK\r\nContent-type: text/html\r\n\r\n')
            cl.send(response)
            cl.close()
        except Exception as e:
            print("Connection error:", e)
            cl.close()

# Main Execution
try:
    serve()
except KeyboardInterrupt:
    print("Stopping Server...")
    led_external.off()
    led_onboard.off()

Hardware Setup

Pico W Pin 20 (GP15): Connect to LED Long Leg (+).

Resistor (220Ω): Place between LED short leg (-) and GND.

GND Pin: Connect to common ground rail.

Important: Use Thonny IDE to save this as 'main.py' on the Pico W.

Software Logic

1. WiFi Station (STA)

The Pico acts as a device on your network. It gets an IP address from your router.

2. The Loop

`s.accept()` waits for a connection. When you visit the IP in your browser, it triggers this loop.

3. Request Parsing

We look for `/on` or `/off` strings inside the text sent by your browser to decide what to do with the GPIO pin.
